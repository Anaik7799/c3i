#### 1. Multi-Level Timeout Management

**System Timeouts**:
- **Patient**: 15 minutes (900 seconds)
- **Smart**: Variable based on strategy
- **Fast**: 10 minutes (600 seconds)
- **Ultra-Fast**: 5 minutes (300 seconds)
- **Selective**: Variable based on scope

**Implementation**:
```elixir
defmodule Indrajaal.CompilationSystem.TimeoutManager do
  def with_timeout(strategy, fun, opts \\ []) do
    timeout = calculate_timeout(strategy, opts)

    task = Task.async(fun)

    case Task.yield(task, timeout) || Task.shutdown(task, :brutal_kill) do
      {:ok, result} ->
        {:ok, result}
      nil ->
        {:timeout, timeout}
    end
  end

  defp calculate_timeout(:patient, opts), do: (opts[:timeout] || 15) * 60 * 1000
  defp calculate_timeout(:fast, opts), do: (opts[:timeout] || 10) * 60 * 1000
  defp calculate_timeout(:ultra_fast, opts), do: (opts[:timeout] || 5) * 60 * 1000
  defp calculate_timeout(:selective, opts) do
    base_timeout = 300_000 # 5 minutes base
    scale_factor = calculate_scale_factor(opts)
    round(base_timeout * scale_factor)
  end
end
```

# lib/indrajaal/compilation_system/progress_tracker.ex
defmodule Indrajaal.CompilationSystem.ProgressTracker do
  @moduledoc """
  Advanced progress tracking for compilation with file-level granularity
  """

  use GenServer

  @type file_status :: :queued | :compiling | :completed | :failed
  @type domain_progress :: %{
    name: String.t(),
    files_total: integer(),
    files_completed: integer(),
    files_failed: integer(),
    percentage: float(),
    estimated_remaining: integer(),
    current_file: String.t() | nil
  }

  @type compilation_state :: %{
    start_time: DateTime.t(),
    total_files: integer(),
    completed_files: integer(),
    failed_files: integer(),
    domain_progress: [domain_progress()],
    file_timings: %{String.t() => integer()},
    memory_usage: [integer()],
    current_phase: String.t()
  }

  def start_compilation_tracking(strategy, opts \\ []) do
    GenServer.start_link(__MODULE__, %{
      strategy: strategy,
      detailed_logs: opts[:detailed_logs] || false,
      progress_per_file: opts[:progress_per_file] || false,
      claude_mode: opts[:claude_mode] || false
    }, name: __MODULE__)
  end

  def track_file_start(file_path) do
    GenServer.cast(__MODULE__, {:file_start, file_path, System.monotonic_time(:millisecond)})
  end

  def track_file_complete(file_path, memory_delta) do
    GenServer.cast(__MODULE__, {:file_complete, file_path, System.monotonic_time(:millisecond), memory_delta})
  end

  def track_file_failed(file_path, error) do
    GenServer.cast(__MODULE__, {:file_failed, file_path, error})
  end

  def get_current_progress do
    GenServer.call(__MODULE__, :get_progress)
  end

  def get_claude_status do
    GenServer.call(__MODULE__, :get_claude_status)
  end

  # Claude-specific status for AI decision making
  def handle_call(:get_claude_status, _from, state) do
    status = %{
      overall_progress: calculate_overall_progress(state),
      critical_issues: identify_critical_issues(state),
      performance_metrics: calculate_performance_metrics(state),
      resource_utilization: get_resource_utilization(),
      recommendations: generate_claude_recommendations(state),
      failure_patterns: analyze_failure_patterns(state),
      estimated_completion: estimate_completion_time(state),
      should_abort: should_recommend_abort?(state),
      optimization_opportunities: identify_optimizations(state)
    }

    {:reply, status, state}
  end

  defp should_recommend_abort?(state) do
    cond do
      state.failed_files > state.total_files * 0.2 ->
        {:yes, "Too many failures (#{state.failed_files}/#{state.total_files})"}

      get_memory_usage() > 7_000 ->
        {:yes, "Memory usage critical (#{get_memory_usage()}MB)"}

      DateTime.diff(DateTime.utc_now(), state.start_time) > 1800 ->
        {:yes, "Compilation taking too long (>30 minutes)"}

      true ->
        {:no, "Compilation proceeding normally"}
    end
  end

  defp generate_claude_recommendations(state) do
    recommendations = []

    # Performance recommendations
    recommendations = if get_memory_usage() > 4000 do
      ["Consider switching to selective compilation to reduce memory pressure" | recommendations]
    else
      recommendations
    end

    # Strategy recommendations
    avg_file_time = calculate_average_file_time(state)
    recommendations = if avg_file_time > 5000 do  # >5 seconds per file
      ["Files compiling slowly - consider ultra_fast strategy" | recommendations]
    else
      recommendations
    end

    # Failure pattern recommendations
    recommendations = if has_dependency_failures?(state) do
      ["Dependency resolution issues detected - may need clean build" | recommendations]
    else
      recommendations
    end

    recommendations
  end
end

defmodule Indrajaal.CompilationSystem.DetailedLogger do
  @moduledoc """
  Comprehensive logging system for compilation with Claude-friendly output
  """

  require Logger

  def log_compilation_start(strategy, opts) do
    log_entry = %{
      event: "compilation_start",
      strategy: strategy,
      timestamp: DateTime.utc_now(),
      system_info: get_system_info(),
      project_info: get_project_info(),
      configuration: get_compilation_config(opts)
    }

    Logger.info("🚀 COMPILATION STARTED", log_entry)
    write_detailed_log(log_entry)
  end

  def log_file_compilation(file_path, phase, details \\ %{}) do
    log_entry = %{
      event: "file_compilation",
      file: file_path,
      phase: phase,
      timestamp: DateTime.utc_now(),
      memory_usage: get_current_memory(),
      details: details
    }

    case phase do
      :start ->
        Logger.debug("📄 Compiling #{Path.basename(file_path)}", log_entry)
      :complete ->
        duration = details[:duration] || 0
        Logger.debug("✅ Completed #{Path.basename(file_path)} (#{duration}ms)", log_entry)
      :failed ->
        Logger.error("❌ Failed #{Path.basename(file_path)}: #{details[:error]}", log_entry)
    end

    write_detailed_log(log_entry)

    # Claude-specific structured output
    if Application.get_env(:indrajaal, :claude_mode, false) do
      emit_claude_event(log_entry)
    end
  end

  def log_domain_progress(domain, progress) do
    log_entry = %{
      event: "domain_progress",
      domain: domain,
      progress: progress,
      timestamp: DateTime.utc_now()
    }

    Logger.info("🏗️  #{domain}: #{progress.completed}/#{progress.total} files (#{Float.round(progress.percentage, 1)}%)", log_entry)
    write_detailed_log(log_entry)
  end

  def log_performance_metrics(metrics) do
    log_entry = %{
      event: "performance_metrics",
      metrics: metrics,
      timestamp: DateTime.utc_now()
    }

    Logger.info("📊 Performance: #{metrics.files_per_minute} files/min, #{metrics.memory_usage}MB memory", log_entry)
    write_detailed_log(log_entry)
  end

  def log_claude_decision_point(decision_type, context, recommendation) do
    log_entry = %{
      event: "claude_decision_point",
      decision_type: decision_type,
      context: context,
      recommendation: recommendation,
      timestamp: DateTime.utc_now()
    }

    Logger.info("🤖 Claude Decision Point: #{decision_type} -> #{recommendation}", log_entry)
    write_detailed_log(log_entry)
    emit_claude_event(log_entry)
  end

  defp emit_claude_event(log_entry) do
    # Emit structured event that Claude can consume
    event_data = %{
      type: "compilation_event",
      data: log_entry,
      claude_context: %{
        can_intervene: log_entry.event in ["file_failed", "performance_degraded", "memory_critical"],
        suggested_actions: generate_suggested_actions(log_entry),
        decision_required: requires_claude_decision?(log_entry)
      }
    }

    # Send to Claude monitoring channel
    Phoenix.PubSub.broadcast(
      Indrajaal.PubSub,
      "claude:compilation:events",
      {:claude_compilation_event, event_data}
    )
  end

  defp write_detailed_log(log_entry) do
    log_file = "tmp/compilation_detailed_#{Date.to_string(Date.utc_today())}.jsonl"
    File.mkdir_p!(Path.dirname(log_file))

    json_line = Jason.encode!(log_entry) <> "\n"
    File.write!(log_file, json_line, [:append])
  end

  defp get_system_info do

  end

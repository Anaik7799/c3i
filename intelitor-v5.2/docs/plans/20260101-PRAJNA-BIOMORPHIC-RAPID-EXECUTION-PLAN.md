### 3.2 Guardian Integration Spec

```elixir
# lib/indrajaal/cockpit/prajna/guardian_integration.ex
defmodule Indrajaal.Cockpit.Prajna.GuardianIntegration do
  @moduledoc """
  Two-way Guardian integration for Prajna commands.

  ## Flow
  1. Prajna.Orchestrator sends command proposal
  2. Guardian.validate/2 checks against envelope + Founder's Directive
  3. Guardian returns {:ok, approved} or {:veto, reason, fallback}
  4. Prajna executes only approved commands

  ## STAMP
  - SC-CONST-007: Guardian absolute veto
  - SC-GDE-001: All mutations require Guardian approval
  """

  @spec submit_proposal(command :: map()) :: {:ok, map()} | {:veto, String.t(), map()}
  def submit_proposal(command) do
    with {:ok, validated} <- Guardian.validate(command, context()),
         {:ok, registered} <- ImmutableRegister.append(validated) do
      {:ok, registered}
    else
      {:veto, reason, fallback} -> {:veto, reason, fallback}
      {:error, reason} -> {:veto, reason, default_fallback()}
    end
  end
end
```

# lib/indrajaal/cockpit/prajna/sentinel_bridge.ex
defmodule Indrajaal.Cockpit.Prajna.SentinelBridge do
  @moduledoc """
  Bidirectional bridge between Prajna SmartMetrics and Sentinel.

  ## Data Flow
  - SmartMetrics → Sentinel: Health observations
  - Sentinel → Prajna: Threat advisories

  ## STAMP
  - SC-IMMUNE-001: Health scoring 0-100
  - SC-IMMUNE-007: Guardian notification for CRITICAL
  """

  use GenServer

  def start_link(opts) do
    GenServer.start_link(__MODULE__, opts, name: __MODULE__)
  end

  @impl true
  def init(_opts) do
    schedule_sync()
    {:ok, %{last_sync: nil}}
  end

  @impl true
  def handle_info(:sync, state) do
    metrics = Prajna.SmartMetrics.current()
    Sentinel.observe(metrics)

    threats = Sentinel.get_active_threats()
    Prajna.Domain.update_advisories(threats)

    schedule_sync()
    {:noreply, %{state | last_sync: DateTime.utc_now()}}
  end

  defp schedule_sync, do: Process.send_after(self(), :sync, 30_000)
end

# Fast OODA loop - 30 second cycle
defmodule Prajna.BiomorphicExecutor do
  @cycle_ms 30_000
  @quality_threshold 0.80

  def run_cycle(state) do
    # Observe (async, non-blocking)
    observations = Task.async_stream([
      &observe_coverage/0,
      &observe_test_results/0,
      &observe_compilation_status/0,
      &observe_api_metrics/0
    ], timeout: 5_000) |> Enum.to_list()

    # Orient (AI with 20ms timeout, fallback to heuristics)
    orientation = case orient_with_ai(observations, 20) do
      {:ok, result} -> result
      :timeout -> orient_heuristic(observations)
    end

    # Decide (with hysteresis - 10% margin, 3-cycle hold)
    decision = decide_with_hysteresis(orientation, state.history)

    # Act (non-blocking execution)
    actions = execute_decision(decision)

    # Update state with learning
    %{state |
      cycle: state.cycle + 1,
      history: [decision | Enum.take(state.history, 9)],
      last_actions: actions
    }
  end
end

### 1.2 Founder's Directive Pattern

```elixir
# Pattern: Validate against Three Supreme Goals
def validate_action(action) do
  goals = [
    {:survival, &check_symbiotic_survival/1, :critical},
    {:sentience, &check_sentience_pursuit/1, :high},
    {:power, &check_power_accumulation/1, :medium}
  ]

  Enum.reduce_while(goals, {:ok, action}, fn {goal, check, priority}, acc ->
    case check.(action) do
      :ok -> {:cont, acc}
      {:violation, reason} when priority == :critical ->
        {:halt, {:reject, "#{goal}: #{reason}"}}
      {:violation, reason} ->
        Logger.warning("#{goal} concern: #{reason}")
        {:cont, acc}
    end
  end)
end
```

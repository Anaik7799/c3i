defmodule WarningElimination.FPPS.Enhanced do
  @moduledoc """
  Enhanced False Positive Prevention System with multi-method validation
  MANDATORY: All fixes must pass consensus validation before application
  """

  @validation_methods [
    :pattern_analysis,
    :ast_structural_analysis,
    :semantic_analysis,
    :dependency_analysis,
    :cross_reference_analysis
  ]

  def validate_fix_proposal(file_path, fix_type, proposed_change) do
    results = Enum.map(@validation_methods, fn method ->
      apply_validation_method(method, file_path, fix_type, proposed_change)
    end)

    case check_consensus(results) do
      {:consensus, :safe} ->
        {:proceed, generate_fix_plan(proposed_change, results)}

      {:consensus, :unsafe} ->
        {:halt, generate_risk_analysis(proposed_change, results)}

      {:no_consensus, disagreements} ->
        {:investigate, generate_disagreement_analysis(disagreements)}
    end
  end

  defp apply_validation_method(:pattern_analysis, file_path, fix_type, change) do
    # Check against known error patterns EP001-EP999
    pattern_db = load_error_patterns()

    case identify_patterns(file_path, fix_type, change, pattern_db) do
      {:safe_pattern, patterns} ->
        {:safe, %{method: :pattern_analysis, patterns: patterns, confidence: 0.9}}

      {:risky_pattern, patterns} ->
        {:unsafe, %{method: :pattern_analysis, risks: patterns, confidence: 0.8}}

      {:unknown_pattern, _} ->
        {:uncertain, %{method: :pattern_analysis, confidence: 0.5}}
    end
  end

  defp apply_validation_method(:ast_structural_analysis, file_path, fix_type, change) do
    # Parse AST and analyze structural impact
    case Code.string_to_quoted(File.read!(file_path)) do
      {:ok, ast} ->
        impact = analyze_ast_impact(ast, change)
        {:safe, %{method: :ast_structural_analysis, impact: impact, confidence: 0.95}}

      {:error, _} ->
        {:unsafe, %{method: :ast_structural_analysis, error: "syntax_error", confidence: 0.99}}
    end
  end

  defp apply_validation_method(:semantic_analysis, file_path, fix_type, change) do
    # Analyze semantic meaning and variable usage context
    context = extract_semantic_context(file_path, change)

    case validate_semantic_consistency(context, change) do
      {:consistent, analysis} ->
        {:safe, %{method: :semantic_analysis, analysis: analysis, confidence: 0.85}}

      {:inconsistent, conflicts} ->
        {:unsafe, %{method: :semantic_analysis, conflicts: conflicts, confidence: 0.9}}
    end
  end

  defp check_consensus(results) do
    safety_votes = Enum.map(results, fn {decision, _data} -> decision end)

    case {Enum.count(safety_votes, &(&1 == :safe)),
          Enum.count(safety_votes, &(&1 == :unsafe)),
          Enum.count(safety_votes, &(&1 == :uncertain))} do

      {safe, 0, 0} when safe == length(@validation_methods) ->
        {:consensus, :safe}

      {0, unsafe, 0} when unsafe == length(@validation_methods) ->
        {:consensus, :unsafe}

      {safe, unsafe, uncertain} when safe > unsafe + uncertain ->
        {:consensus, :safe}

      {safe, unsafe, uncertain} when unsafe > safe + uncertain ->
        {:consensus, :unsafe}

      _ ->
        {:no_consensus, results}
    end
  end
end

defmodule WarningElimination.VariableResolver do
  @moduledoc """
  Systematic resolution of variable naming conflicts
  Addresses both undefined variables and unused variable warnings
  """

  def resolve_variable_conflicts(file_path) do
    content = File.read!(file_path)

    # Parse and analyze all variable usage
    usage_analysis = analyze_variable_usage(content)

    # Identify conflicts between underscore prefixes and actual usage
    conflicts = identify_conflicts(usage_analysis)

    # Generate resolution plan
    resolution_plan = create_resolution_plan(conflicts, usage_analysis)

    # Apply fixes with validation
    apply_resolutions(file_path, resolution_plan)
  end

  defp analyze_variable_usage(content) do
    # Extract all variable definitions and usages
    definitions = extract_variable_definitions(content)
    usages = extract_variable_usages(content)

    # Cross-reference to identify actual usage patterns
    Enum.map(definitions, fn {var_name, contexts} ->
      actual_usages = Map.get(usages, var_name, [])

      %{
        variable: var_name,
        defined_contexts: contexts,
        used_contexts: actual_usages,
        is_actually_used: length(actual_usages) > 0,
        has_underscore_prefix: String.starts_with?(var_name, "_"),
        conflict_type: determine_conflict_type(var_name, actual_usages)
      }
    end)
  end

  defp determine_conflict_type(var_name, usages) do
    cond do
      String.starts_with?(var_name, "_") and length(usages) > 0 ->
        :underscore_but_used

      not String.starts_with?(var_name, "_") and length(usages) == 0 ->
        :no_underscore_but_unused

      true ->
        :no_conflict
    end
  end

  defp create_resolution_plan(conflicts, usage_analysis) do
    Enum.map(conflicts, fn conflict ->
      case conflict.conflict_type do
        :underscore_but_used ->
          %{
            action: :remove_underscore,
            variable: conflict.variable,
            new_name: String.trim_leading(conflict.variable, "_"),
            justification: "Variable is actually used, underscore causes undefined variable error"
          }

        :no_underscore_but_unused ->
          %{
            action: :add_underscore,
            variable: conflict.variable,
            new_name: "_#{conflict.variable}",
            justification: "Variable is defined but never used"
          }

        :no_conflict ->
          %{action: :no_change, variable: conflict.variable}
      end
    end)
    |> Enum.filter(fn plan -> plan.action != :no_change end)
  end
end

defmodule WarningElimination.BatchProcessor do
  @batch_size 25  # Reduced from 100 for better validation
  @validation_required true

  def process_warning_batches(all_warnings) do
    all_warnings
    |> Enum.chunk_every(@batch_size)
    |> Enum.with_index(1)
    |> Enum.reduce_while({:ok, []}, fn {batch, batch_num}, {status, results} ->
      Logger.info("🔧 Processing batch #{batch_num}/#{ceil(length(all_warnings) / @batch_size)}")

      case process_single_batch(batch, batch_num) do
        {:ok, batch_result} ->
          # Mandatory compilation validation after each batch
          case validate_compilation_after_batch(batch_num) do
            {:ok, _} ->
              {:cont, {:ok, [batch_result | results]}}

            {:error, compilation_errors} ->
              Logger.error("❌ Batch #{batch_num} caused compilation errors. Rolling back.")
              rollback_batch(batch_num)
              {:halt, {:error, {:compilation_failure, batch_num, compilation_errors}}}
          end

        {:error, reason} ->
          Logger.error("❌ Batch #{batch_num} processing failed: #{inspect(reason)}")
          {:halt, {:error, {:batch_failure, batch_num, reason}}}
      end
    end)
  end

  defp process_single_batch(warnings, batch_num) do
    # Create git checkpoint before batch
    create_batch_checkpoint(batch_num)

    # Process warnings in batch with FPPS validation
    batch_results = Enum.map(warnings, fn warning ->
      case WarningElimination.FPPS.Enhanced.validate_fix_proposal(
        warning.file,
        warning.type,
        warning.proposed_fix
      ) do
        {:proceed, fix_plan} ->
          apply_validated_fix(warning, fix_plan)

        {:halt, risk_analysis} ->
          Logger.warn("⚠️ Fix halted for safety: #{inspect(risk_analysis)}")
          {:skipped, warning, risk_analysis}

        {:investigate, disagreement} ->
          Logger.warn("🔍 Fix requires investigation: #{inspect(disagreement)}")
          {:investigate, warning, disagreement}
      end
    end)

    # Analyze batch results
    {successful, failed, skipped} = categorize_batch_results(batch_results)

    Logger.info("📊 Batch #{batch_num} results: #{length(successful)} successful, #{length(failed)} failed, #{length(skipped)} skipped")

    {:ok, %{
      batch_number: batch_num,
      successful: successful,
      failed: failed,
      skipped: skipped,
      total_processed: length(warnings)
    }}
  end

  defp validate_compilation_after_batch(batch_num) do
    Logger.info("🔍 Validating compilation after batch #{batch_num}")

    case System.cmd("mix", ["compile", "--warnings-as-errors"], stderr_to_stdout: true) do
      {_output, 0} ->
        Logger.info("✅ Compilation successful after batch #{batch_num}")
        {:ok, :compilation_success}

      {error_output, _} ->
        Logger.error("❌ Compilation failed after batch #{batch_num}")
        parse_compilation_errors(error_output)
    end
  end

  defp create_batch_checkpoint(batch_num) do
    checkpoint_message = """
    checkpoint: before batch #{batch_num} processing

    Checkpoint Details:
    - Batch Number: #{batch_num}
    - Timestamp: #{DateTime.utc_now()}
    - Compilation Status: Validated before batch
    - Git SHA: #{get_current_commit_sha()}

    SOPv5.11 Status:
    - Agent Coordination: Active
    - FPPS Validation: Enabled
    - TPS Methodology: Applied

    Recovery Plan:
    - Rollback command: git reset --hard checkpoint-batch-#{batch_num}
    - Validation required after rollback

    Checkpoint-ID: CB-#{batch_num}-#{:os.system_time(:millisecond)}
    """

    System.cmd("git", ["add", "-A"])
    System.cmd("git", ["commit", "-m", checkpoint_message])
    System.cmd("git", ["tag", "-a", "checkpoint-batch-#{batch_num}", "-m", "Checkpoint before batch #{batch_num}"])
  end

  defp analyze_file_for_warning_patterns(file_path) do

    # Check for common warning-inducing patterns
    potential_issues = [
      check_unused_variable_patterns(content),
      check_unreachable_code_patterns(content),
      check_module_attribute_patterns(content),
      check_function_usage_patterns(content)
    ]
    |> List.flatten()
    |> Enum.filter(& &1)

    if length(potential_issues) > 0 do
      Logger.warn("⚠️ Potential warning patterns detected in #{file_path}")
      create_prevention_notification(file_path, potential_issues)
    end
  end

  defp check_unused_variable_patterns(content) do
    # Look for variables that are defined but potentially not used
    variable_definitions = Regex.scan(~r/(\w+)\s*=/, content)
    variable_usages = extract_variable_usages_from_content(content)

    Enum.filter(variable_definitions, fn [_, var_name] ->
      # Check if variable is defined but not used
      usage_count = count_variable_usage(var_name, variable_usages)
      usage_count == 0 and not String.starts_with?(var_name, "_")
    end)
    |> Enum.map(fn [_, var_name] ->
      %{
        type: :unused_variable_risk,
        variable: var_name,
        suggestion: "Consider prefixing with underscore if truly unused: _#{var_name}",
        confidence: 0.7
      }
    end)
  end

      file: file_path,

    #{if outcome, do: "## Outcome\n#{outcome}", else: ""}
    """

    # Link to related decisions
    related = find_related_decisions(decision)

    Client.capture_knowledge(:executive, content,
      tags: ["decision", decision.category],
      links: related,
      emotional_weight: decision.importance / 10.0
    )
  end

  @doc """
  Capture stream of consciousness from Black Box.
  Encrypted at rest, only Twin can access.
  """
  def capture_consciousness(stream) do
    content = """
    # Stream of Consciousness

    #{stream.transcription}

    ## Detected Themes
    #{Enum.join(stream.themes, ", ")}

    ## Emotional State
    #{stream.emotional_state}
    """

    Client.capture_knowledge(:black_box, content,
      tags: ["consciousness", "private"],
      privacy: :black_box_,
      emotional_weight: stream.intensity
    )
  end

  defp format_options(options) do
    options
    |> Enum.with_index(1)
    |> Enum.map(fn {opt, i} -> "#{i}. #{opt}" end)
    |> Enum.join("\n")
  end

  defp find_related_decisions(decision) do
    {:ok, results} = Client.search_knowledge(decision.title, modules: [:executive], limit: 5)
    Enum.map(results, & &1["id"])
  end
end

defmodule Indrajaal.Chaya.Integration.ServiceRouter do
  @moduledoc """
  Central router for all Indrajaal service integrations.
  Routes Chaya module requests to appropriate Indrajaal services.

  ## STAMP Constraints
  - SC-BIO-001: OODA cycle < 100ms
  - SC-PRAJNA-001: Guardian pre-approval for mutations
  - SC-ZENOH-001: Zenoh telemetry for all service calls
  """

  use GenServer
  alias Indrajaal.{Guardian, Sentinel, Zenoh}
  alias Indrajaal.Chaya.Holon.{State, Register}
  alias Indrajaal.Chaya.SMRITI.Client, as: SMRITI

  @ooda_timeout 100  # ms

  # Service category routing
  @service_routes %{
    # Category 1: Core Infrastructure
    config: Indrajaal.Config,
    feature_flags: Indrajaal.FeatureFlags,

    # Category 2: Biomorphic & Immune
    sentinel: Indrajaal.Sentinel,
    guardian: Indrajaal.Guardian,
    pattern_hunter: Indrajaal.Sentinel.PatternHunter,
    symbiotic_defense: Indrajaal.Sentinel.SymbioticDefense,

    # Category 3: Observability
    telemetry: Indrajaal.Telemetry,
    logger: Indrajaal.Quadplex.Logger,
    metrics: Indrajaal.Metrics,

    # Category 4: State Management
    holon_state: Indrajaal.Chaya.Holon.State,
    register: Indrajaal.Chaya.Holon.Register,
    smriti: Indrajaal.Chaya.SMRITI.Client,

    # Category 5: Communication
    zenoh: Indrajaal.Zenoh,
    pubsub: Phoenix.PubSub,
    event_bus: Indrajaal.EventBus,

    # Category 6: Security
    proof_token: Indrajaal.Guardian.ProofToken,
    ucan: Indrajaal.UCAN,
    vault: Indrajaal.Vault,

    # Category 7: Agents
    agent_swarm: Indrajaal.Chaya.Symbiotic.AgentSwarm,
    ooda: Indrajaal.CAE.OODAController,

    # Category 8: Knowledge & AI
    openrouter: Indrajaal.Cortex.OpenRouter,
    umbra: Indrajaal.Chaya.UmbraEngine,

    # Category 9: Integration
    mcp: Indrajaal.MCP.Server,
    external_api: Indrajaal.Integration.ExternalAPI
  }

  def start_link(opts) do
    GenServer.start_link(__MODULE__, opts, name: __MODULE__)
  end

  @doc """
  Route a Chaya module request to the appropriate Indrajaal service.
  Includes Guardian approval for mutations and Zenoh telemetry.
  """
  def route(module, action, params, opts \\ []) do
    start_time = System.monotonic_time(:microsecond)

    with :ok <- validate_ooda_budget(start_time),
         {:ok, service} <- resolve_service(module, action),
         {:ok, proof} <- maybe_get_approval(action, opts),
         {:ok, result} <- call_service(service, action, params),
         :ok <- publish_telemetry(module, action, result, start_time) do
      {:ok, result}
    end
  end

  defp validate_ooda_budget(start_time) do
    elapsed = System.monotonic_time(:microsecond) - start_time
    if elapsed < @ooda_timeout * 1000 do
      :ok
    else
      {:error, :ooda_budget_exceeded}
    end
  end

  defp resolve_service(module, action) do
    case Map.get(@service_routes, service_key(module, action)) do
      nil -> {:error, :unknown_service}
      service -> {:ok, service}
    end
  end

  defp maybe_get_approval(action, opts) do
    if is_mutation?(action) do
      Guardian.request_approval(%{
        holon: :chaya,
        action: action,
        actor: Keyword.get(opts, :actor, :twin)
      })
    else
      {:ok, nil}
    end
  end

  defp publish_telemetry(module, action, result, start_time) do
    elapsed = System.monotonic_time(:microsecond) - start_time

    Zenoh.publish("chaya/telemetry/service_call", %{
      module: module,
      action: action,
      result: result_status(result),
      duration_us: elapsed,
      timestamp: DateTime.utc_now()
    })

    :ok
  end
end

defmodule Indrajaal.Chaya.SMRITI.SemanticDB do
  @moduledoc """
  SMRITI as the universal semantic database for Chaya.
  Replaces Neo4j (graph) and TimescaleDB (time-series).

  ## Capabilities
  - Graph queries via [[wiki-links]] and backlinks
  - Time-series via DuckDB OLAP with ASOF joins
  - Semantic search via vector embeddings
  - Knowledge graph via Zettelkasten atomic notes
  - MCP endpoints for AI agent access

  ## STAMP Constraints
  - SC-SMRITI-001: All memories as Zettels
  - SC-HOLON-001: SQLite/DuckDB only for state
  - SC-SMRITI-004: MCP endpoints for AI access
  """

  alias Indrajaal.Chaya.SMRITI.Client

  @doc """
  Graph query using backlinks (replaces Neo4j Cypher).
  """
  def graph_query(start_zettel_id, opts \\ []) do
    depth = Keyword.get(opts, :depth, 2)
    direction = Keyword.get(opts, :direction, :both)

    # Get backlinks recursively up to depth
    traverse_graph(start_zettel_id, depth, direction, MapSet.new())
  end

  @doc """
  Time-series query using DuckDB ASOF join (replaces TimescaleDB).
  """
  def time_series_query(module, metric, opts \\ []) do
    start_time = Keyword.get(opts, :start, DateTime.add(DateTime.utc_now(), -7, :day))
    end_time = Keyword.get(opts, :end, DateTime.utc_now())
    granularity = Keyword.get(opts, :granularity, :hour)

    # DuckDB OLAP query for time-series aggregation
    """
    SELECT
      time_bucket('#{granularity}', timestamp) as bucket,
      avg(value) as avg_value,
      min(value) as min_value,
      max(value) as max_value
    FROM history.duckdb
    WHERE module = '#{module}'
      AND metric = '#{metric}'
      AND timestamp BETWEEN '#{start_time}' AND '#{end_time}'
    GROUP BY bucket
    ORDER BY bucket
    """
    |> DuckDBex.query()
  end

  @doc """
  Semantic search with entropy weighting (replaces vector DB).
  """
  def semantic_search(query, opts \\ []) do
    limit = Keyword.get(opts, :limit, 10)
    modules = Keyword.get(opts, :modules, [])

    # SMRITI handles vector embeddings internally
    {:ok, results} = Client.search_knowledge(query, modules: modules, limit: limit * 2)

    # Weight by freshness (1 - entropy) for better relevance
    results
    |> Enum.map(fn r ->
      freshness = 1.0 - Map.get(r, "entropy", 0.5)
      score = Map.get(r, "score", 0.5) * freshness
      Map.put(r, "weighted_score", score)
    end)
    |> Enum.sort_by(& &1["weighted_score"], :desc)
    |> Enum.take(limit)
  end

  @doc """
  Knowledge graph cluster analysis (replaces Neo4j community detection).
  """
  def cluster_analysis(opts \\ []) do
    min_connections = Keyword.get(opts, :min_connections, 5)

    # Get all Zettels with their backlink counts
    {:ok, graph} = Client.get_knowledge_graph()

    # Simple clustering based on backlink density
    graph["nodes"]
    |> Enum.filter(fn node ->
      length(get_backlinks(node["id"])) >= min_connections
    end)
    |> Enum.group_by(& &1["cluster"])
  end

  defp traverse_graph(zettel_id, 0, _direction, visited), do: visited
  defp traverse_graph(zettel_id, depth, direction, visited) do
    if MapSet.member?(visited, zettel_id) do
      visited
    else
      visited = MapSet.put(visited, zettel_id)

      neighbors = case direction do
        :outgoing -> get_forward_links(zettel_id)
        :incoming -> get_backlinks(zettel_id)
        :both -> get_forward_links(zettel_id) ++ get_backlinks(zettel_id)
      end

      Enum.reduce(neighbors, visited, fn neighbor_id, acc ->
        traverse_graph(neighbor_id, depth - 1, direction, acc)
      end)
    end
  end
end
